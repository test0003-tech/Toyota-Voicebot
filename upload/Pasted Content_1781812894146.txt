function toggleSim(){

function randName(){return NAMES[nameIdx++%NAMES.length];}
function randPhone(){return PHONES[Math.floor(Math.random()*PHONES.length)];}
function randScript(key,name){const a=SCRIPTS[key];return a[Math.floor(Math.random()*a.length)].replace('{name}',name.split(' ')[0]);}

function addNewLead(){
  const name=randName(), phone=randPhone();
  const id='lead-'+Date.now()+Math.random();
  const outcome=Math.random()<0.35?'interested':Math.random()<0.5?'callback':Math.random()<0.5?'junk':'notpicked';
  const lead={id,name,phone,outcome,attempt:1,status:'calling'};
  leads.push(lead);
  renderAgg();
  totalCalls++;
  activeCalls++;
  hourlyData[Math.floor(Math.random()*8)]++;
  renderCallCard(lead);
  updateCounters();
  // simulate call duration
  const dur=5000+Math.random()*6000;
  setTimeout(()=>finishCall(lead,dur),dur);
}

function finishCall(lead,dur){

  if(lead.outcome==='interested'){

  } else if(lead.outcome==='junk'){
    lead.status='junk'; junkCount++;
    addToJunk(lead);
  } else if(lead.outcome==='callback'){
    lead.status='callback';
  } else {
    lead.status='notpicked';
  }
  renderAgg();
  updateCardStatus(lead);
  updateCounters();
  updateAnalytics();
}

  document.getElementById('kpi-active').textContent=A;
  const conv=T>0?Math.round((C/T)*100):0;
  document.getElementById('kpi-conv-rate').textContent=conv+'% conversion';
  document.getElementById('kpi-junk-pct').textContent=(T>0?Math.round((J/T)*100):0)+'% of total';
  // funnel
  const pct=n=>T>0?Math.round((n/T)*100):0;
  setFunnel('total',T,100);
  setFunnel('connected',connected,pct(connected));
  setFunnel('yes',yes,pct(yes));
  setFunnel('crm',C,pct(C));
  setFunnel('junk2',J,pct(J));
  // donut
  const circ=251;
  const gi=T>0?C/T:0, gj=T>0?J/T:0, ga=T>0?A/T:0;
  setDonutArc('d-interested',gi,0,circ);
  setDonutArc('d-junk',gj,gi,circ);
  setDonutArc('d-active',ga,gi+gj,circ);
  document.getElementById('d-pct').textContent=(T>0?Math.round((C/T)*100):0)+'%';
  document.getElementById('dl-interested').textContent=`Interested (${C})`;
  document.getElementById('dl-junk').textContent=`Junk (${J})`;
  document.getElementById('dl-active').textContent=`In Progress (${A})`;
  // bars
  renderBars();
  // attempt rates (simulated)
  if(T>0){
    document.getElementById('ar-1').textContent=Math.round(55+Math.random()*10)+'%';
    document.getElementById('ar-2').textContent=Math.round(40+Math.random()*10)+'%';
    document.getElementById('ar-3').textContent=Math.round(28+Math.random()*8)+'%';
    document.getElementById('ar-4').textContent=Math.round(18+Math.random()*7)+'%';
    document.getElementById('ar-wa').textContent=Math.round(22+Math.random()*10)+'%';
  }
}

function setFunnel(key,val,pct){
  const fill=document.getElementById('f-'+key);
  const cnt=document.getElementById('fc-'+key);
  if(fill){fill.style.width=Math.max(pct,2)+'%';fill.textContent=val>0?val:'';}
  if(cnt) cnt.textContent=val;
}

function setDonutArc(id,frac,offset,circ){
  const el=document.getElementById(id);
  if(!el) return;
  const dash=Math.round(frac*circ);
  const off=Math.round(offset*circ);
  el.setAttribute('stroke-dasharray',dash+' '+(circ-dash));
  el.setAttribute('stroke-dashoffset',63-off);
}

function renderBars(){
  const bc=document.getElementById('bar-chart');
  bc.innerHTML='';
  const max=Math.max(...hourlyData,1);
  const hours=['9am','10am','11am','12pm','1pm','2pm','3pm','4pm'];
  hourlyData.forEach((v,i)=>{
    const h=Math.round((v/max)*100);
    bc.innerHTML+=`<div class="bar-col"><div class="bar-val">${v}</div><div class="bar" style="height:${h}%;background:var(--gold);opacity:${0.5+i*0.07}"></div><div class="bar-label">${hours[i]}</div></div>`;
  });
}
