   753→function updateAnalytics(){
   754→  const T=leads.length, C=crmLeads.length, J=junkLeads.length, A=activeCalls;
   755→  const connected=Math.round(T*0.7), yes=C;
   756→  document.getElementById('kpi-total').textContent=T;

   758→  document.getElementById('kpi-junk').textContent=J;
   759→  document.getElementById('kpi-active').textContent=A;
   760→  const conv=T>0?Math.round((C/T)*100):0;
   761→  document.getElementById('kpi-conv-rate').textContent=conv+'% conversion';
   762→  document.getElementById('kpi-junk-pct').textContent=(T>0?Math.round((J/T)*100):0)+'% of total';
   763→  // funnel
   764→  const pct=n=>T>0?Math.round((n/T)*100):0;
   765→  setFunnel('total',T,100);
   766→  setFunnel('connected',connected,pct(connected));
   767→  setFunnel('yes',yes,pct(yes));
   768→  setFunnel('crm',C,pct(C));
   769→  setFunnel('junk2',J,pct(J));
   770→  // donut
   771→  const circ=251;
   772→  const gi=T>0?C/T:0, gj=T>0?J/T:0, ga=T>0?A/T:0;
   773→  setDonutArc('d-interested',gi,0,circ);
   774→  setDonutArc('d-junk',gj,gi,circ);
   775→  setDonutArc('d-active',ga,gi+gj,circ);
   776→  document.getElementById('d-pct').textContent=(T>0?Math.round((C/T)*100):0)+'%';
   777→  document.getElementById('dl-interested').textContent=`Interested (${C})`;
   778→  document.getElementById('dl-junk').textContent=`Junk (${J})`;
   779→  document.getElementById('dl-active').textContent=`In Progress (${A})`;
   780→  // bars
   781→  renderBars();
   782→  // attempt rates (simulated)
   783→  if(T>0){
   784→    document.getElementById('ar-1').textContent=Math.round(55+Math.random()*10)+'%';
   785→    document.getElementById('ar-2').textContent=Math.round(40+Math.random()*10)+'%';
   786→    document.getElementById('ar-3').textContent=Math.round(28+Math.random()*8)+'%';
   787→    document.getElementById('ar-4').textContent=Math.round(18+Math.random()*7)+'%';
   788→    document.getElementById('ar-wa').textContent=Math.round(22+Math.random()*10)+'%';
   789→  }
   790→}

   799→function setDonutArc(id,frac,offset,circ){

   805→  el.setAttribute('stroke-dashoffset',63-off);
